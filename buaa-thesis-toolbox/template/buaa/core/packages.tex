%% English logo: negative kern + alternating raise/lower.
%% Usage: \englogo{HELLO}  — keep short; output is an unbreakable \mbox.
\ExplSyntaxOn
\int_new:N \l__buaa_englogo_int
\NewDocumentCommand \englogo { m }
  {
    \mbox
      {
        \group_begin:
        \int_zero:N \l__buaa_englogo_int
        \buaa_englogo:n {#1}
        \group_end:
      }
  }
\cs_new_protected:Npn \buaa_englogo:n #1
  {
    \tl_map_inline:nn {#1}
      {
        \token_if_eq_meaning:NNTF ##1 \c_space_token
          {
            \c_space_tl
            \int_zero:N \l__buaa_englogo_int
          }
          {
            \int_incr:N \l__buaa_englogo_int
            \int_compare:nT { \l__buaa_englogo_int > 1 } { \kern -.12em }
            \int_case:nnF { \int_mod:nn { \l__buaa_englogo_int } { 3 } }
              {
                { 1 } { \hbox {##1} }
                { 2 }
                  {
                    {
                      \fontsize {0.8em} {\baselineskip} \selectfont
                      \raise .47ex \hbox {##1}
                    }
                  }
                { 0 } { \lower .5ex \hbox {##1} }
              }
              { \hbox {##1} }
          }
      }
  }
\ExplSyntaxOff

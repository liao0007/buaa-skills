%% buaa/core/helpers.tex — shared conditionals and font helpers
\newcommand{\buaa@ifsMastersThesis}[2]{%
  \ifthenelse{\equal{\@thesis}{master} \OR \equal{\@thesis}{professional}}{#1}{#2}%
}
\newcommand{\buaa@ifsDoctorThesis}[2]{%
  \ifthenelse{\equal{\@thesis}{doctor} \OR \equal{\@thesis}{prodoctor}}{#1}{#2}%
}
\newcommand{\buaa@latin}{\rmfamily\upshape}

\newcommand{\buaa@setupLatinFont}{%
  \ifpdftex
    \ifthenelse{\equal{\@ostype}{win}}{\RequirePackage{times}}{}
    \ifthenelse{\equal{\@ostype}{linux}}{%
      \RequirePackage{newtxtext}%
      \RequirePackage{newtxmath}%
    }{}
    \ifthenelse{\equal{\@ostype}{mac}}{\RequirePackage{times}}{}
  \else
    \IfFontExistsTF{Times New Roman}{%
      \setmainfont{Times New Roman}[%
        BoldFont = * Bold,
        ItalicFont = * Italic,
        BoldItalicFont = * Bold Italic
      ]%
    }{%
      \ClassWarning{buaa}{%
        Times New Roman not found; using TeX Gyre Termes instead.%
      }%
      \setmainfont{TeX Gyre Termes}[%
        Extension = .otf,
        UprightFont = *-regular,
        BoldFont = *-bold,
        ItalicFont = *-italic,
        BoldItalicFont = *-bolditalic
      ]%
    }%
  \fi
}

\RequirePackage{iftex}
\buaa@setupLatinFont

\newcommand{\clearautopage}{%
  \clearpage
  \ifthenelse{\equal{\@printtype}{twoside}}{%
    \ifodd\c@page
    \else
      \hbox{}
      \vspace*{\fill}
      \begin{center}
        {\textcolor[rgb]{0.75,0.75,0.75}{\@emptypagewords}}
      \end{center}
      \vspace{\fill}
      \thispagestyle{empty}
      \newpage
      \if@twocolumn \hbox{} \newpage \fi
    \fi
  }{}
}

\newcommand{\buaa@resetAppendixCounters}{%
  \setcounter{section}{0}%
  \setcounter{subsection}{0}%
  \setcounter{subsubsection}{0}%
  \gdef\thesection{\@Alph\c@section}%
  \gdef\thesubsection{\@Alph\c@section.\@arabic\c@subsection}%
  \renewcommand{\thefigure}{A\arabic{figure}}%
  \setcounter{figure}{0}%
  \renewcommand{\thetable}{A\arabic{table}}%
  \setcounter{table}{0}%
  \renewcommand{\theequation}{A\arabic{equation}}%
  \setcounter{equation}{0}%
}
\newcommand{\buaa@beginUnnumberedChapter}[1]{%
  \ctexset{chapter/numbering=false}%
  \chapter{#1}%
  \buaa@resetAppendixCounters
}

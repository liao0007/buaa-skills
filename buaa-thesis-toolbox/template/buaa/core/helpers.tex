%% buaa/core/helpers.tex — shared conditionals and font helpers
\newcommand{\buaa@ifsMastersThesis}[2]{%
  \ifthenelse{\equal{\buaa@thesis}{master} \OR \equal{\buaa@thesis}{professional}}{#1}{#2}%
}
\newcommand{\buaa@ifsDoctorThesis}[2]{%
  \ifthenelse{\equal{\buaa@thesis}{doctor} \OR \equal{\buaa@thesis}{prodoctor}}{#1}{#2}%
}
\newcommand{\buaa@latin}{\rmfamily\upshape}

\newcommand{\buaa@setupFonts}{%
  \IfFileExists{buaa/font/setup.tex}{%
    \input{buaa/font/setup.tex}%
  }{%
    \ClassError{buaa}{Missing buaa/font/setup.tex}{%
      Add buaa/font/setup.tex with XeLaTeX/LuaLaTeX font definitions.%
    }%
  }%
}

\buaa@setupFonts
\AddToHook{begindocument/before}{\buaa@setMathFont}

\newcommand{\clearautopage}{%
  \clearpage
  \ifthenelse{\equal{\@printtype}{print}}{%
    \ifodd\c@page
    \else
      \hbox{}
      \vspace*{\fill}
      \begin{center}
        {\textcolor[rgb]{0.75,0.75,0.75}{\@emptypagewords}}
      \end{center}
      \vspace{\fill}
      \thispagestyle{empty}
      \newpage
      \if@twocolumn \hbox{} \newpage \fi
    \fi
  }{}
}

\newcommand{\emptypage}{%
  \clearpage
  \hbox{}
  \vspace*{\fill}
  \begin{center}
    {\textcolor[rgb]{0.75,0.75,0.75}{\@emptypagewords}}
  \end{center}
  \vspace{\fill}
  \thispagestyle{empty}
  \addtocounter{page}{-1}
  \newpage
  \if@twocolumn \hbox{} \newpage \fi
}

\newcommand{\buaa@resetAppendixCounters}{%
  \setcounter{section}{0}%
  \setcounter{subsection}{0}%
  \setcounter{subsubsection}{0}%
  \gdef\thesection{\@Alph\c@section}%
  \gdef\thesubsection{\@Alph\c@section.\@arabic\c@subsection}%
  \renewcommand{\thefigure}{A\arabic{figure}}%
  \setcounter{figure}{0}%
  \renewcommand{\thetable}{A\arabic{table}}%
  \setcounter{table}{0}%
  \renewcommand{\theequation}{A\arabic{equation}}%
  \setcounter{equation}{0}%
}
\newcommand{\buaa@beginUnnumberedChapter}[1]{%
  \ctexset{chapter/numbering=false}%
  \chapter{#1}%
  \buaa@resetAppendixCounters
}

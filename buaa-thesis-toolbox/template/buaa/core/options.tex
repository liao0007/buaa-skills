%% Output profile. Both official profiles use mirrored headers; print additionally
%% inserts blank verso pages between major parts.
\DeclareOption{library}{\gdef\@printtype{library}\PassOptionsToClass{twoside,openany}{ctexbook}}
\DeclareOption{print}{\gdef\@printtype{print}\PassOptionsToClass{twoside,openany}{ctexbook}}
\DeclareOption{Library}{\gdef\@printtype{library}\PassOptionsToClass{twoside,openany}{ctexbook}}
\DeclareOption{Print}{\gdef\@printtype{print}\PassOptionsToClass{twoside,openany}{ctexbook}}
\DeclareOption{printtype}{\ClassWarning{buaa}{%
  You have not specified the PRINTTYPE option. Use library(default) or print.}}

%%
%% AutoFakeBold: SimSun has no real Bold; without this, \songti\bfseries looks regular.
%% Times lock: Pandoc injects lmodern which would otherwise steal Latin to LM (tight caps).

\newcommand{\buaa@setLatinFont}{%
  \setmainfont{Times New Roman}[
    Path=./buaa/font/,
    Extension=.ttf,
    UprightFont=Times New Roman,
    BoldFont=Times New Roman Bold,
    ItalicFont=Times New Roman Italic,
    BoldItalicFont=Times New Roman Bold Italic,
  ]%
}
\buaa@setLatinFont
%% Re-apply after Pandoc's \usepackage{lmodern} (loads after class in .tex).
\AddToHook{package/lmodern/after}{\buaa@setLatinFont}
\AtBeginDocument{\buaa@setLatinFont}

\newfontfamily\SimSunPageNum{simsun}[
  Path=./buaa/font/,
  Extension=.ttc,
]

\setCJKmainfont{simsun.ttc}[
  Path=./buaa/font/,
  AutoFakeBold=4,

\setCJKfamilyfont{zhkai}{simkai.ttf}[Path=./buaa/font/]

\gdef\buaa@setMathFont{%
  \IfFileExists{buaa/font/Cambria Math.ttf}{%
    \setmathfont{Cambria Math}[Path=./buaa/font/,Extension=.ttf]%
  }{%
    \ClassWarningNoLine{buaa}{Cambria Math is missing; using STIX Two Math as the rendering fallback}%
    \setmathfont{STIX Two Math}%
  }%
}
